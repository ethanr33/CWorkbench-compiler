<root> ::= <function>
<function> ::= <type> "IDENT" "(" ")" "{" <return> "}"
<type> ::= "INT"
<return> ::= "RETURN" <returnval> ";"
<literal> ::= "INT_CONST"
<binaryoptype> ::= "+"
<returnval> ::= <literal> <binaryop>
<binaryop> ::= <binaryoptype> <binaryopsuffix> | "ε"
<binaryopsuffix> ::= <literal> <binaryop>
