<root> ::= <function>
<function> ::= <type> "IDENT" "(" ")" "{" <funcbody> "}"
<type> ::= "INT"
<return> ::= "RETURN" <returnval> ";"
<returnval> ::= <arithmeticexpr> | "IDENT"
<arithmeticexpr> ::= "INT_CONST" <arithmeticexprsuffix>
<arithmeticexprsuffix> ::= "+" <arithmeticexpr> | "ε"
<statement> ::= <type> "IDENT" "=" "INT_CONST" ";"
<funcbody> ::= <funcbodyprefix> <return>
<funcbodyprefix> ::= <statement> <funcbodyprefix> | "ε"
