<root> ::= <function>
<function> ::= <type> "IDENT" "(" ")" "{" <functionbody> "}"
<type> ::= "INT"
<literal> ::= "INT_CONST"
<value> ::= <literal> | "IDENT"
<binaryoptype> ::= "+" | "-" | "*" | "/" | "="
<functionbody> ::= <statement> <functionbody> | "ε"
<statement> ::= <return> | <variabledeclaration> | <expression> ";"
<return> ::= "RETURN" <expression> ";"
<variabledeclaration> ::= <type> "IDENT" <variabledefinition> ";"
<variabledefinition> ::= "=" <expression> | "ε"
<expression> ::= <exprval> <binaryop>
<exprval> ::= "(" <expression> ")" | <value>
<binaryop> ::= <binaryoptype> <expression> | "ε"
